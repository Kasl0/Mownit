using Polynomials
using Plots

# Wielomiany Legendre'a
function P(k)
    if k == 0
        return Polynomial([1])
    end
    if k == 1
        return Polynomial([0,1])
    end
    return (2*k - 1) / k * Polynomial([0,1]) * P(k-1) - (k-1) / k * P(k-2)
end

# Rysujemy wykresy dla wielomianw do 5 stopnia
num = 5

plot()
for k in 0:num
    xs = range(-1, 1, 100)
    ys = P(k).(xs)
    plot!(xs, ys, label="P$k(x)")
end

xlabel!("x")
ylabel!("Pk(x)")
title!("Legendre Polynomials")

using Polynomials
using QuadGK

function integral(k, f)
    (xp, a) = gauss(Float64, k)
    sum(a .* f.(xp)) 
end

function integral2(a, b, k, f)
    F(z) = f((b + a) / 2 + (b - a) / 2 * z) * (b - a) / 2
    integral(k, F)
end

Zadanie 5 (3p)¶

Napisz wasne funkcje cakujce - w wybranym przedziale (a,b) - zoonymi metodami prostoktw, trapezw oraz Simpsona. Dla staego przedziau zwikszaj liczb uytych punktw (n). Narysuj wykresy funkcji bdu w stosunku do wyniku otrzymanego analitycznie, w zalenosci od $h=\frac{b-a}{n}$:

  1. wykadniczej
  2. trygonometrycznej (np. sin (x) w przedziale $[0,2\pi]$)
In [ ]:
function rectangle(f, a, b, n)
    h = (b - a) / n
    sum = 0
    for i in 0:n-1
        sum += h * f(a + (i+0.5)*h)
    end
    return sum
end

function trapeze(f, a, b, n)
    h = (b - a) / n
    sum = 0
    for i in 0:n-1
        sum += h * (f(a + i*h) + f(a + (i+1)*h)) / 2
    end
    return sum
end

function simpson(f, a, b, n)
    h = (b - a) / n
    sum = 0
    for i in 0:n-1
        sum += h / 6 * (f(a + i*h) + f(a + (i+1)*h) + 4*f(a + (i+0.5)*h))
    end
    return sum
end
